 bool isAnagram(string s, string t) {
        int count[256] = { 0 };
        int m = s.size();
        int n = t.size();

        if(m!=n) return false;

        for(int i = 0 ; s[i] && t[i] ; i++){
            count[s[i]]++;
            count[t[i]]--;
        }

        for (int i = 0; i < 256; i++)
        if (count[i])
            return false;
    return true;
    }
