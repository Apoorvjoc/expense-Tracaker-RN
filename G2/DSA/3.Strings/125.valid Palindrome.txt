 bool isValid(char ch){
        if((ch >= 65 && ch<91)||(ch >= 97 && ch<123)||(ch>=48 && ch<=57)){
            return true;
        }
        return false;
    }   

    bool isPalindrome(string s) {
        int l = 0 , r = s.size()-1;
        transform(s.begin(), s.end(), s.begin(), ::tolower);
        cout<<s<<" : ";
        while(l<=r){
            if(s[l] == s[r]){
                l++ ; r--;
            }else if(isspace(s[l])){
                l++;
            }else if(isspace(s[r])){
                r--;
            }else if(!isValid(s[l])){
                l++;
            }else if(!isValid(s[r])){
                r--;
            }else if(s[l]!=s[r]){
                return false;
            }
        }
       if(l>=r){
           return true;
       }else{
           return false;
       }
    }
