 int countNegatives(vector<vector<int>>& grid) {
        int cnt = 0;
        int r = grid.size();
        int c = grid[0].size();
        int sc = 0 , sr = 0;
        for(int i = 0 ; i<r*c ; i++){
            if(sc == c){
                sr++;
                sc = 0;
            }
            if(grid[sr][sc++] < 0){
                cnt++;
            }
        }
        return cnt;
    }
